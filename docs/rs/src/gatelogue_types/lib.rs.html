1//! Rust utility library for using Gatelogue data in Rust projects. It will load the database for you to access via ORM or raw SQL.
2//!
3//! # Installation
4//! Add to your `Cargo.toml`:
5//! ```toml
6//! gatelogue-types = { version = "3", features = [...] }
7//!
8//! # To import directly from the repository:
9//! gatelogue-types = { git = "https://github.com/mrt-map/gatelogue", package = "gatelogue-types", features = [...] }
10//! ```
11//! Features as denoted by `...` include `reqwest_get`, `surf_get`, `ureq_get` (to download with `reqwest`, `surf` and `ureq` respectively) and `bundled` (to bundle SQLite).
12//!
13//! # Usage
14//! To retrieve the data:
15//! ```rust
16//! use gatelogue_types::{GD, getter};
17//! # #[tokio::main]
18//! # async fn main() -> Result<(), Box<dyn std::error::Error>> {
19//! // with pre-written getter!() functions:
20//! let gd = GD::get_async_no_sources(getter!(reqwest)).await?; // retrieve data (async)
21//! let gd = GD::get_no_sources(getter!(ureq))?; // retrieve data (blocking)
22//!
23//! // or with any HTTP client of your choice (with a callable that takes in a &str and returns an AsRef<[u8]>):
24//! let gd = GD::get_async_no_sources(async |url| surf::get(url).recv_bytes().await); // retrieve data (async)
25//! let gd = GD::get_no_sources(|url| ureq::get(url).call()?.into_body().read_to_vec())?; // retrieve data (blocking)
26//!
27//! // similar syntax can be used if you want the sources, with `get_async_with_sources` and `get_with_sources`
28//! # Ok(()) }
29//! ```
30//! `getter!()` accepts the following inputs:
31//! - `reqwest` (async) requiring `reqwest`
32//! - `reqwest_blocking` (blocking) requiring `reqwest` with `blocking` feature
33//! - `surf` (async) requiring `surf`
34//! - `ureq` (blocking) requiring `ureq`
35//! - `isahc` (blocking) requiring `isahc`
36//! - `isahc_async` (async) requiring `isahc`
37//! - `attohttpc` (blocking) requiring `attohttpc` (untested)
38//! - `minreq` (blocking) requiring `minreq`
39//! - `wreq` (async) requiring `wreq`
40//!
41//! Using the ORM does not require SQL and makes for generally clean code.
42//! However, doing this is very inefficient as each attribute access is one SQL query.
43//! ```rust
44//! use gatelogue_types::AirAirport;
45//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
46//! # let gd = gatelogue_types::GD::get_no_sources(gatelogue_types::getter!(ureq))?;
47//! for airport in gd.nodes_of_type::<AirAirport>()? {
48//!     for gate in airport.gates(&gd)? {
49//!         println!("Airport {:?} has gate {:?}", airport.code(&gd)?, gate.code(&gd)?)
50//!     }
51//! }
52//! # Ok(()) }
53//! ```
54//!
55//! Querying the underlying `SQLite` database directly with `rusqlite` is generally more efficient and faster.
56//! It is also the only way to access the `*Source` tables, if you retrieved the database with those.
57//!
58//! ```rust
59//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
60//! # let gd = gatelogue_types::GD::get_no_sources(gatelogue_types::getter!(ureq))?;
61//! gd.0.prepare("SELECT A.code, G.code FROM AirGate G INNER JOIN AirAirport A ON G.airport = A.i")?
62//!     .query_map((), |row| {
63//!         println!("Airport {:?} has gate {:?}", row.get::<_, String>(0)?, row.get::<_, String>(1)?);
64//!         Ok(())
65//!     })?;
66//! # Ok(()) }
67//! ```

84pub const URL: &str =
85    "https://raw.githubusercontent.com/MRT-Map/gatelogue/refs/heads/dist/data.db";
86pub const URL_NO_SOURCES: &str =
87    "https://raw.githubusercontent.com/MRT-Map/gatelogue/refs/heads/dist/data-ns.db";

89#[macro_export]
90macro_rules! getter {
91    (reqwest) => {async |url: &'static str| -> Result<Vec<u8>, reqwest::Error> {Ok(reqwest::get(url).await?.bytes().await?.to_vec())}};
92    (reqwest_blocking) => {|url: &'static str| -> Result<Vec<u8>, reqwest::Error> {Ok(reqwest::blocking::get(url)?.bytes()?.to_vec())}};
93    (surf) => {async |url: &'static str| -> Result<Vec<u8>, surf::Error> {surf::get(url).recv_bytes().await}};
94    (ureq) => {|url: &'static str| -> Result<Vec<u8>, ureq::Error> {ureq::get(url).call()?.into_body().read_to_vec()}};
95    (isahc) => {|url: &'static str| -> Result<Vec<u8>, isahc::Error> {Ok(isahc::ReadResponseExt::bytes(&mut isahc::get(url)?)?)}};
96    (isahc_async) => {async |url: &'static str| -> Result<Vec<u8>, isahc::Error> {Ok(isahc::AsyncReadResponseExt::bytes(&mut isahc::get_async(url).await?).await?)}};
97    (attohttpc) => {|url: &'static str| -> Result<Vec<u8>, attohttpc::Error> {attohttpc::get(url).send()?.bytes()}};
98    (minreq) => {|url: &'static str| -> Result<Vec<u8>, minreq::Error> {Ok(minreq::get(url).send()?.into_bytes())}};
99    (wreq) => {async |url: &'static str| -> Result<Vec<u8>, wreq::Error> {Ok(wreq::get(url).send().await?.bytes().await?.to_vec())}};
100}

102pub struct GD(pub Connection);

104impl GD {
105    fn from_bytes(bytes: &[u8]) -> Result<Self> {
106        let mut conn = Connection::open_in_memory()?;
107        conn.deserialize_read_exact("main", bytes, bytes.len(), true)?;
108        Ok(Self(conn))
109    }
110    pub async fn get_async_with_sources<F: AsyncFnOnce(&'static str) -> Result<B, E>, B: AsRef<[u8]>, E: Debug + Send + Sync + 'static>(getter: F) -> Result<Self> {
111        Self::from_bytes(getter(URL).await.map_err(|e| Error::HTTPGetError(Box::new(e)))?.as_ref())
112    }
113    pub async fn get_async_no_sources<F: AsyncFnOnce(&'static str) -> Result<B, E>, B: AsRef<[u8]>, E: Debug + Send + Sync + 'static>(getter: F) -> Result<Self> {
114        Self::from_bytes(getter(URL_NO_SOURCES).await.map_err(|e| Error::HTTPGetError(Box::new(e)))?.as_ref())
115    }
116    pub fn get_with_sources<F: FnOnce(&'static str) -> Result<B, E>, B: AsRef<[u8]>, E: Debug + Send + Sync + 'static>(getter: F) -> Result<Self> {
117        Self::from_bytes(getter(URL).map_err(|e| Error::HTTPGetError(Box::new(e)))?.as_ref())
118    }
119    pub fn get_no_sources<F: FnOnce(&'static str) -> Result<B, E>, B: AsRef<[u8]>, E: Debug + Send + Sync + 'static>(getter: F) -> Result<Self> {
120        Self::from_bytes(getter(URL_NO_SOURCES).map_err(|e| Error::HTTPGetError(Box::new(e)))?.as_ref())
121    }
122
123    pub fn timestamp(&self) -> Result<String> {
124        self.0
125            .query_one("SELECT timestamp FROM Metadata", (), |a| a.get(0))
126            .map_err(Into::into)
127    }
128    pub fn version(&self) -> Result<u32> {
129        self.0
130            .query_one("SELECT version FROM Metadata", (), |a| a.get(0))
131            .map_err(Into::into)
132    }
133    pub fn has_sources(&self) -> Result<bool> {
134        self.0
135            .query_one("SELECT has_sources FROM Metadata", (), |a| a.get(0))
136            .map_err(Into::into)
137    }
138
139    pub fn nodes(&self) -> Result<Vec<AnyNode>> {
140        self.0
141            .query_and_then_get_vec("SELECT i FROM Node", (), |a| {
142                AnyNode::from_id(self, a.get(0)?).map(|a| a.unwrap())
143            })
144    }
145    pub fn located_nodes(&self) -> Result<Vec<AnyLocatedNode>> {
146        self.0
147            .query_and_then_get_vec("SELECT i FROM LocatedNode", (), |a| {
148                AnyLocatedNode::from_id(self, a.get(0)?).map(|a| a.unwrap())
149            })
150    }
151    pub fn nodes_of_type<T: Node + From<ID> + FromSql>(&self) -> Result<Vec<T>> {
152        let ty = T::from(0).ty();
153        self.0.query_and_then_get_vec(
154            "SELECT i FROM Node WHERE type = ?",
155            (ty,),
156            |a| Ok(a.get(0)?),
157        )
158    }
159    pub fn get_node(&self, id: ID) -> Result<Option<AnyNode>> {
160        AnyNode::from_id(self, id)
161    }
162    pub fn get_located_node(&self, id: ID) -> Result<Option<AnyLocatedNode>> {
163        AnyLocatedNode::from_id(self, id)
164    }
165}
